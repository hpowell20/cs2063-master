try {
            appContext.openFileOutput(fileName, Context.MODE_PRIVATE).use {

            }
        } catch (e: Exception) {
            when (e) {
                is FileNotFoundException -> Log.e(TAG, "File not found: " + e.message)
                is IOException -> Log.e(TAG, "IO Exception: " + e.message)
                else -> e.printStackTrace()
            }
        }
